# Shared VPN pod. Owns the network + user namespace that gluetun and ANY
# number of other containers can ride through — not just qBittorrent.
#
# UserNS=keep-id lives HERE (once), not on individual member containers:
# rootless Podman rejects UserNS=keep-id directly on a container that also
# does Network=container:<other> (verified: status=126, container never
# created). Pod-level keep-id sidesteps that — member containers just join
# via Pod=gluetun.pod and inherit it, so their PUID/PGID-based file ownership
# (bind-mounted config/downloads) still maps to your real host UID.
#
# To route a FUTURE container's traffic through this VPN:
#   1. In its <app>.container, add `Pod=gluetun.pod` and remove any
#      `PublishPort=`/`Network=` of its own (the pod owns networking now).
#   2. Add that app's port to PublishPort= below.
#   3. Add `Requires=gluetun.service` / `After=gluetun.service` to its [Unit]
#      (so it doesn't start before the tunnel container exists).
# No UserNS= needed on the new container — it inherits the pod's.
[Pod]
PodName=gluetun
UserNS=keep-id

# One line per app that rides this VPN. qBittorrent today; add more here as
# you connect future containers (e.g. PublishPort=9117:9117 for a future app).
PublishPort=8080:8080
